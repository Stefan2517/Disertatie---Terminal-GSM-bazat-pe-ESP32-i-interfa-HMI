#define TX_PIN 14
#define RX_PIN 27

void setup() {
  Serial.begin(115200);
  // Inițializăm Serial2 pe pinii noi
  Serial2.begin(9600, SERIAL_8N1, RX_PIN, TX_PIN);
  
  delay(1000);
  Serial.println("--- TEST NOU PINI 14/27 ---");
}

void loop() {
  // Trimitem AT automat la fiecare 3 secunde
  Serial.println("Trimit: AT...");
  Serial2.println("AT");
  
  delay(500); // Așteptăm un pic răspunsul
  
  // Citim ce vine de la SIM900
  while (Serial2.available()) {
    char c = Serial2.read();
    Serial.print(c); // Ar trebui să vedem "OK" aici
  }
  
  delay(2500);
}
