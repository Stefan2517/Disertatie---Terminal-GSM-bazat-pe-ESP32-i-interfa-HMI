#include <Arduino.h>

#define RX_GSM 27
#define TX_GSM 14
#define RX_NEXTION 32
#define TX_NEXTION 33

String rawMsg = "";
String gsmBuffer = "";
String smsSender = ""; 
bool asteaptaContinut = false; 

void proceseazaComanda(String data);
void handleIncomingSMS(String header, String content);
void handleIncomingCall(String line);
void updateOperator();
void trimiteLaNextion(String cmd);

void setup() {
  Serial.begin(115200);
  Serial2.begin(9600, SERIAL_8N1, RX_GSM, TX_GSM);
  Serial1.begin(9600, SERIAL_8N1, RX_NEXTION, TX_NEXTION);

  delay(4000);
  Serial2.println("AT"); 
  delay(500);
  Serial2.println("AT+CMGF=1"); 
  delay(200);
  Serial2.println("AT+CSCS=\"GSM\""); 
  delay(200);
  Serial2.println("AT+CNMI=2,2,0,0,0"); 
  delay(200);
  Serial2.println("AT+CLIP=1"); 
  
  updateOperator();
}

void loop() {
  // Procesare comenzi de la Nextion
  while (Serial1.available()) {
    char c = Serial1.read();
    if (isPrintable(c) || (uint8_t)c == 0xFF) rawMsg += c;
    if (rawMsg.endsWith("\xFF\xFF\xFF")) {
      String cleanMsg = rawMsg.substring(0, rawMsg.length() - 3);
      proceseazaComanda(cleanMsg);
      rawMsg = ""; 
    }
  }

  // Procesare date de la GSM (State Machine)
  while (Serial2.available()) {
    char g = Serial2.read();
    gsmBuffer += g;
    
    if (g == '\n') {
      gsmBuffer.trim();
      
      if (gsmBuffer.length() > 0) {
        // Logica SMS
        if (gsmBuffer.indexOf("+CMT:") != -1) {
          int f = gsmBuffer.indexOf("\"");
          int s = gsmBuffer.indexOf("\"", f + 1);
          smsSender = (f != -1 && s != -1) ? gsmBuffer.substring(f + 1, s) : "Unknown";
          asteaptaContinut = true;
        } 
        else if (asteaptaContinut) {
          handleIncomingSMS(smsSender, gsmBuffer);
          asteaptaContinut = false;
        }
        // Logica Apeluri
        else if (gsmBuffer.indexOf("+CLIP") != -1) {
          handleIncomingCall(gsmBuffer);
        }
        else if (gsmBuffer.indexOf("NO CARRIER") != -1 || gsmBuffer.indexOf("BUSY") != -1) {
          trimiteLaNextion("page page0");
          updateOperator();
        }
      }
      gsmBuffer = ""; 
    }
  }
}

void proceseazaComanda(String data) {
  data.trim();
  if (data.length() < 1) return;

  // Comanda de închidere (pentru butonul de pe Nextion)
  if (data == "close") {
    trimiteLaNextion("page page0");
    return;
  }

  if (data.indexOf("ATH") != -1) {
    Serial2.println("ATH");
    trimiteLaNextion("page page0");
    return;
  }
  
  if (data.startsWith("s")) {
    int virgula = data.indexOf(",");
    if (virgula != -1) {
      String numar = data.substring(1, virgula);
      String mesaj = data.substring(virgula + 1);
      numar.replace("\"", "");
      numar.trim();
      Serial2.print("AT+CMGS=\""); Serial2.print(numar); Serial2.println("\"");
      delay(500);
      Serial2.print(mesaj);
      Serial2.write(26); 
    }
    return;
  }
  
  if (data.startsWith("p")) {
    String numar = data.substring(1);
    Serial2.print("ATD"); Serial2.print(numar); Serial2.println(";");
    trimiteLaNextion("page page2");
  }
}

void handleIncomingSMS(String expeditor, String content) {
  trimiteLaNextion("page page6"); 
  delay(100);
  trimiteLaNextion("page6.t0.txt=\"" + expeditor + "\"");
  trimiteLaNextion("page6.t1.txt=\"" + content + "\"");
}

void handleIncomingCall(String line) {
  int f = line.indexOf("\"");
  int s = line.indexOf("\"", f + 1);
  if (f != -1 && s != -1) {
    String numar = line.substring(f + 1, s);
    trimiteLaNextion("page page5");
    trimiteLaNextion("page5.t1.txt=\"" + numar + "\"");
  }
}

void updateOperator() {
  Serial2.println("AT+COPS=0,0");
  delay(500);
  Serial2.println("AT+COPS?");
}

void trimiteLaNextion(String cmd) {
  Serial1.print(cmd);
  Serial1.write(0xff); Serial1.write(0xff); Serial1.write(0xff);
}
